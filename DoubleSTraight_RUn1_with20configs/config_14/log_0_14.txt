def identifyCornerRegions(corners):

    right_corners = []

    ys = corners[:, 1]

    min_y, max_y = min(ys), max(ys)
    
    for corner in corners:
        x, y = corner
        region = []

            region.append('left')
        elif x == max_x:
            region.append('right')
        if y == min_y:
            region.append('top')

            region.append('bottom')

            left_corners.append(corner)

def identifyPickandPlace(corners, center):

    top_right_corner = None

            top_right_corner = max(corner_list, key=lambda c: c[0])
    
    pick_point = top_right_corner
    place_point = center
    
    return pick_point, place_point

def findDistanceAndDirection(pick_point, place_point):
    dx = place_point[0] - pick_point[0]
    dy = place_point[1] - pick_point[1]
    distance = np.sqrt(dx**2 + dy**2)
